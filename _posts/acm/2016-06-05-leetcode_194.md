```bash
row=$(sed -n '$=' file.txt)
col=$(awk 'NR==1{print NF}' file.txt)
awk -v row=$row -v col=$col '{ for(i=1;i<=NF;i++)a[NR"-"i]=$i  }END{ for(i=1;i<=col;i++){ for(j=1;j<=row;j++) if (j==row){ printf("%s",a[j"-"i]); }else{ printf("%s ",a[j"-"i]);  }print "" }  }' file.txt
