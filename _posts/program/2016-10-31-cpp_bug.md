```cpp
#include<iostream>
#include<cstring>
#include<string>
using namespace std;

class Test(){
    public:
      Test(const char* p){
        int size = strlen(p);
        ptr = new char[size + 1];
        ptr[size]='\0';
        strcpy(ptr,p);
        cout<<"constructor!"<<endl;
      }
      ~Test(){
        delete ptr;
        cout<<"destructor"<<endl;
      }
    public:
     char* ptr=nullptr;
};


Test getTest(){
  Test t(1);
  return t;
}

int main(void){
  const char* p = getTest()->ptr;
  cout<<p<<endl;
  return 0;
}
```
